class Solution:
    def calPoints(self, ops: List[str]) -> int:
        stack = []
        for c in ops:
            if c == "+":
                if len(stack) > 1:
                    stack.append(stack[-1]+stack[-2])
            elif c == "D":

            else:
                stack.append(int(c))
